event_days = []
features = []
ids = []
targets = []

for _, row in raw_df.iterrows():
    feature = row.codes
    pid = row.PID
    target = row.target
    
    for day in range(len(row.to_event)):
        event_days.append(row.to_event[day])
        features.append(feature[day])
        ids.append(pid)
        targets.append(target)

long_event = pd.DataFrame({
    'PID': ids,
    'feature': features,
    'event_day': event_days,
    'target': targets
})
